# {
# ckey: zuhayr
# character_name: Jane Doe

# item_name: ugly plush toy
# item_icon: flagmask     
# item_desc: It's truly hideous.
# req_titles: Assistant, Security Officer
# req_access: 1
# paper_title: Instruction
# paper_info: Do not throw it
# papep_stamp_type: /obj/item/weapon/stamp
# }
